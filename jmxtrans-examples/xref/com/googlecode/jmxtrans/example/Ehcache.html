23  package com.googlecode.jmxtrans.example;
24  
25  import com.google.inject.Injector;
26  import com.googlecode.jmxtrans.JmxTransformer;
27  import com.googlecode.jmxtrans.cli.JmxTransConfiguration;
28  import com.googlecode.jmxtrans.guice.JmxTransModule;
29  import com.googlecode.jmxtrans.model.JmxProcess;
30  import com.googlecode.jmxtrans.model.Query;
31  import com.googlecode.jmxtrans.model.Server;
32  import com.googlecode.jmxtrans.model.output.GraphiteWriter;
33  
34  /**
35   * This example shows how to query ehcache for its statistics information.
36   * 
37   * @author jon
38   */
39  @SuppressWarnings({"squid:S106", "squid:S1118"}) // using StdOut if fine in an example
40  public class Ehcache {
41  
42  	@SuppressWarnings("squid:S1313") // this is an example, no need to make it configurable
43  	private static final String GW_HOST = "192.168.192.133";
44  	private static final JsonPrinter printer = new JsonPrinter(System.out);
45  
46  	/** */
47  	public static void main(String[] args) throws Exception {
48  
49  		JmxProcess process = new JmxProcess(Server.builder()
50  				.setHost("w2")
51  				.setPort("1099")
52  				.setAlias("w2_ehcache_1099")
53  				.addQuery(Query.builder()
54  						.setObj("net.sf.ehcache:CacheManager=net.sf.ehcache.CacheManager@*,name=*,type=CacheStatistics")
55  						.addAttr("CacheHits")
56  						.addAttr("InMemoryHits")
57  						.addAttr("OnDiskHits")
58  						.addAttr("CacheMisses")
59  						.addAttr("ObjectCount")
60  						.addAttr("MemoryStoreObjectCount")
61  						.addAttr("DiskStoreObjectCount")
62  						.addOutputWriterFactory(GraphiteWriter.builder()
63  								.addTypeName("name")
64  								.setDebugEnabled(true)
65  								.setHost(GW_HOST)
66  								.setPort(2003)
67  								.build())
68  						.build())
69  				.build());
70  
71  		printer.prettyPrint(process);
72  
73  		Injector injector = JmxTransModule.createInjector(new JmxTransConfiguration());
74  		JmxTransformer transformer = injector.getInstance(JmxTransformer.class);
75  
76  		transformer.executeStandalone(process);
77  	}
78  }
