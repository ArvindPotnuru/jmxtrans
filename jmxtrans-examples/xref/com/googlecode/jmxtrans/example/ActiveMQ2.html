23  package com.googlecode.jmxtrans.example;
24  
25  import com.google.inject.Injector;
26  import com.googlecode.jmxtrans.JmxTransformer;
27  import com.googlecode.jmxtrans.cli.JmxTransConfiguration;
28  import com.googlecode.jmxtrans.guice.JmxTransModule;
29  import com.googlecode.jmxtrans.model.JmxProcess;
30  import com.googlecode.jmxtrans.model.Query;
31  import com.googlecode.jmxtrans.model.Server;
32  import com.googlecode.jmxtrans.model.output.RRDToolWriter;
33  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
34  
35  import java.io.File;
36  import java.io.IOException;
37  
38  /**
39   * This example shows how to query an ActiveMQ server for some information.
40   * 
41   * The point of this example is to show that * works as part of the objectName.
42   * It also shows that you don't have to set an attribute to get for a query.
43   * jmxtrans will get all attributes on an object if you don't specify any.
44   * 
45   * @author jon
46   */
47  @SuppressWarnings({"squid:S106", "squid:S1118"}) // using StdOut if fine in an example
48  public class ActiveMQ2 {
49  
50  	private static final JsonPrinter jsonPrinter = new JsonPrinter(System.out);
51  
52  	@SuppressFBWarnings(
53  			value = "DMI_HARDCODED_ABSOLUTE_FILENAME",
54  			justification = "Path to RRD binary is hardcoded as this is example code")
55  	public static void main(String[] args) throws Exception {
56  		File outputFile = new File("target/w2-TEST.rrd");
57  		if (!outputFile.exists() && !outputFile.createNewFile()) {
58  			throw new IOException("Could not create output file");
59  		}
60  		RRDToolWriter gw = RRDToolWriter.builder()
61  				.setTemplateFile(new File("memorypool-rrd-template.xml"))
62  				.setOutputFile(outputFile)
63  				.setBinaryPath(new File("/opt/local/bin"))
64  				.setDebugEnabled(true)
65  				.setGenerate(true)
66  				.addTypeName("Destination")
67  				.build();
68  
69  		JmxProcess process = new JmxProcess(Server.builder()
70  				.setHost("w2")
71  				.setPort("1105")
72  				.setAlias("w2_activemq_1105")
73  				.addQuery(Query.builder()
74  						.setObj("org.apache.activemq:BrokerName=localhost,Type=Queue,Destination=*")
75  						.addAttr("QueueSize")
76  						.addAttr("MaxEnqueueTime")
77  						.addAttr("MinEnqueueTime")
78  						.addAttr("AverageEnqueueTime")
79  						.addAttr("InFlightCount")
80  						.addAttr("ConsumerCount")
81  						.addAttr("ProducerCount")
82  						.addAttr("DispatchCount")
83  						.addAttr("DequeueCount")
84  						.addAttr("EnqueueCount")
85  						.addOutputWriterFactory(gw)
86  						.build())
87  				.addQuery(Query.builder()
88  						.setObj("org.apache.activemq:BrokerName=localhost,Type=Topic,Destination=*")
89  						.addAttr("QueueSize")
90  						.addAttr("MaxEnqueueTime")
91  						.addAttr("MinEnqueueTime")
92  						.addAttr("AverageEnqueueTime")
93  						.addAttr("InFlightCount")
94  						.addAttr("ConsumerCount")
95  						.addAttr("ProducerCount")
96  						.addAttr("DispatchCount")
97  						.addAttr("DequeueCount")
98  						.addAttr("EnqueueCount")
99  						.addOutputWriterFactory(gw)
100 						.build()).build());
101 		jsonPrinter.prettyPrint(process);
102 
103 		Injector injector = JmxTransModule.createInjector(new JmxTransConfiguration());
104 		JmxTransformer transformer = injector.getInstance(JmxTransformer.class);
105 		transformer.executeStandalone(process);
106 	}
107 }
