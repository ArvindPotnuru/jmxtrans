23  package com.googlecode.jmxtrans.model.output.support.pool;
24  
25  import com.github.rholder.retry.Attempt;
26  import com.github.rholder.retry.RetryListener;
27  import com.github.rholder.retry.Retryer;
28  import com.github.rholder.retry.RetryerBuilder;
29  import com.github.rholder.retry.StopStrategies;
30  import com.github.rholder.retry.WaitStrategies;
31  import org.slf4j.Logger;
32  import org.slf4j.LoggerFactory;
33  import stormpot.Allocator;
34  import stormpot.Poolable;
35  import stormpot.Slot;
36  
37  import javax.annotation.Nonnull;
38  import java.util.concurrent.Callable;
39  import java.util.concurrent.TimeUnit;
40  
41  public class RetryingAllocator<V extends Poolable> implements Allocator<V> {
42  
43  	private static final Logger log = LoggerFactory.getLogger(RetryingAllocator.class);
44  
45  	/**
46  	 * Interval to wait between allocation attempts
47  	 */
48  	private static final long WAIT_INTERVAL_MS = 100;
49  
50  	/**
51  	 * Max time to wait when attempting to allocate a slot
52  	 */
53  	private static final long MAX_WAIT_MS = TimeUnit.SECONDS.toMillis(60);
54  
55  	@Nonnull private final Allocator<V> allocator;
56  	@Nonnull private final Retryer<V> retryer;
57  
58  	public RetryingAllocator(Allocator<V> allocator) {
59  		this(allocator, WAIT_INTERVAL_MS, MAX_WAIT_MS);
60  	}
61  
62  	/**
63  	 * @param allocator Allocator
64  	 * @param waitInterval Wait time between allocation attempts (ms)
65  	 * @param maxWaitInterval Max time to wait for an allocation (ms)
66  	 */
67  	public RetryingAllocator(Allocator<V> allocator, long waitInterval, long maxWaitInterval) {
68  		this.allocator = allocator;
69  		this.retryer = RetryerBuilder.<V>newBuilder()
70  			.retryIfException()
71  			.withWaitStrategy(WaitStrategies.fixedWait(waitInterval, TimeUnit.MILLISECONDS))
72  			.withStopStrategy(StopStrategies.stopAfterDelay(maxWaitInterval, TimeUnit.MILLISECONDS))
73  			.withRetryListener(new RetryListener() {
74  				@Override
75  				public <V> void onRetry(Attempt<V> attempt) {
76  					if (attempt.hasException()) {
77  						log.error("Error allocating slot", attempt.getExceptionCause());
78  					}
79  				}
80  			})
81  			.build();
82  	}
83  
84  	@Override
85  	public V allocate(final Slot slot) throws Exception {
86  		return retryer.call(new Callable<V>() {
87  			@Override
88  			public V call() throws Exception {
89  				return allocator.allocate(slot);
90  			}
91  		});
92  	}
93  
94  	@Override
95  	public void deallocate(V poolable) throws Exception {
96  		allocator.deallocate(poolable);
97  	}
98  }
