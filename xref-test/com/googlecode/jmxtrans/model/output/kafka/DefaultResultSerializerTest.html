40  public class DefaultResultSerializerTest {
41  	private ObjectMapper objectMapper = new ObjectMapper();
42  
43  	/**
44  	 * Create result with non null epoch
45       */
46  	private static Result numericResultAt(long timestamp) {
47  		Result result = numericResult();
48  		return new Result(timestamp, result.getAttributeName(), result.getClassName(), result.getObjDomain(), result.getKeyAlias(), result.getTypeName(), result.getValues());
49  	}
50  	@Test
51  	public void convertSingleNumericToString() throws Exception {
52  		ImmutableMap<String, String> tags = ImmutableMap.of("myTagKey1", "myTagValue1");
53  		ResultSerializer resultSerializer = new DefaultResultSerializer(ImmutableList.<String>of(), false, "rootPrefix", tags, asList("typeName.type", "className"));
54  
55  		long now = System.currentTimeMillis();
56  		Collection<String> messages = resultSerializer.serialize(dummyServer(), dummyQuery(), numericResultAt(now));
57  
58  		assertThat(messages).hasSize(1);
59  		String message = messages.iterator().next();
60  		// Check JSON syntax
61  		assertThat(message).endsWith("}}");
62  		JsonNode jsonNode = objectMapper.readValue(message, JsonNode.class);
63  		assertThat(jsonNode.get("keyspace").asText()).isEqualTo("rootPrefix.host_example_net_4321.MemoryAlias.ObjectPendingFinalizationCount");
64  		assertThat(jsonNode.get("value").asLong()).isEqualTo(10L);
65  		assertThat(jsonNode.get("timestamp").asLong()).isEqualTo(now / 1000);
66  		assertThat(jsonNode.get("tags").get("myTagKey1").asText()).isEqualTo("myTagValue1");
67  		assertThat(jsonNode.get("tags").get("typeName.type").asText()).isEqualTo("Memory");
68  		assertThat(jsonNode.get("tags").get("className").asText()).endsWith("MemoryImpl");
69  	}
70  
71  	@Test
72  	public void convertSingleNonNumericToString() throws Exception {
73  		ImmutableMap<String, String> tags = ImmutableMap.of("myTagKey1", "myTagValue1");
74  		ResultSerializer resultSerializer = new DefaultResultSerializer(ImmutableList.<String>of(), false, "rootPrefix", tags, ImmutableList.<String>of());
75  
76  		Collection<String> messages = resultSerializer.serialize(dummyServer(), dummyQuery(), stringResult());
77  
78  		assertThat(messages).isEmpty();
79  	}

81  	@Test
82  	public void convertHashToStrings() throws Exception {
83  		ImmutableMap<String, String> tags = ImmutableMap.of("myTagKey1", "myTagValue1");
84  		ResultSerializer resultSerializer = new DefaultResultSerializer(ImmutableList.<String>of(), false, "rootPrefix", tags, ImmutableList.<String>of());
85  
86  		Collection<String> messages = resultSerializer.serialize(dummyServer(), dummyQuery(), hashResult());
87  
88  		assertThat(messages).hasSize(4);
89  		for (String message : messages) {
90  			assertThat(message)
91  					.contains("\"keyspace\":\"rootPrefix.host_example_net_4321.MemoryAlias.NonHeapMemoryUsage_")
92  					.contains("\"value\":")
93  					.contains("\"timestamp\":0")
94  					.contains("\"tags\":{\"myTagKey1\":\"myTagValue1\"");
95  
96  		}
97  	}

104 		assertThat(resultSerializer.getTypeNames()).isEmpty();

107 	}
