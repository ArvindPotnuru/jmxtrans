90  	@Test
91  	public void testEmptyTagSetting() throws Exception {
92  		Map<String, String> tagMap = newHashMap();
93  		Map<String, Object> settings = newHashMap();
94  		settings.put("tags", tagMap);
95  		OpenTSDBGenericWriter writer = new TestOpenTSDBGenericWriter(
96  				ImmutableList.<String>of(),
97  				false,
98  				settings);
99  
100 		Mockito.when(this.mockResult.getAttributeName()).thenReturn("X-ATT-X");
101 		Mockito.when(this.mockResult.getValue()).thenReturn("120021");
102 
103 		// Verify empty tag map.
104 		Assertions.assertThat(writer.getTypeNames()).isEmpty();
105 
106 		writer.start();
107 		writer.doWrite(ServerFixtures.dummyServer(), this.mockQuery, ImmutableList.of(this.mockResult));
108 		writer.close();
109 
110 		Assert.assertTrue(
111 				this.tvMetricLinesSent.get(0).matches("^X-DOMAIN.PKG.CLASS-X\\.X-ATT-X 0 120021 host=[^ ]*$"));
112 	}

114 	@Test
115 	public void testTagSetting() throws Exception {
116 		Map<String, String> tagMap;
117 
118 		// Verify tag map with multiple values.
119 		tagMap = newHashMap();
120 		tagMap.put("x-tag1-x", "x-tag1val-x");
121 		tagMap.put("x-tag2-x", "x-tag2val-x");
122 		tagMap.put("x-tag3-x", "x-tag3val-x");
123 		OpenTSDBGenericWriter writer = createWriter("tags", tagMap);
124 
125 		writer.start();
126 		writer.doWrite(ServerFixtures.dummyServer(), this.mockQuery, ImmutableList.of(this.mockResult));
127 		writer.close();
128 
129 		Assert.assertTrue(this.tvMetricLinesSent.get(0).matches("^X-DOMAIN.PKG.CLASS-X\\.X-ATT-X 0 120021.*"));
130 		Assert.assertTrue(this.tvMetricLinesSent.get(0).matches(".*\\bhost=.*"));
131 		Assert.assertTrue(this.tvMetricLinesSent.get(0).matches(".*\\bx-tag1-x=x-tag1val-x\\b.*"));
132 		Assert.assertTrue(this.tvMetricLinesSent.get(0).matches(".*\\bx-tag2-x=x-tag2val-x\\b.*"));
133 		Assert.assertTrue(this.tvMetricLinesSent.get(0).matches(".*\\bx-tag3-x=x-tag3val-x\\b.*"));
134 	}

154 	@Test
155 	public void testHooksCalled() throws Exception {
156 		OpenTSDBGenericWriter writer = createWriter();
157 		writer.start();
158 		Assert.assertTrue(prepareSenderCalled);
159 		Assert.assertFalse(shutdownSenderCalled);
160 		Assert.assertFalse(startOutputCalled);
161 		Assert.assertFalse(finishOutputCalled);
162 
163 		writer.doWrite(ServerFixtures.dummyServer(), this.mockQuery, ImmutableList.of(this.mockResult));
164 		Assert.assertTrue(prepareSenderCalled);
165 		Assert.assertFalse(shutdownSenderCalled);
166 		Assert.assertTrue(startOutputCalled);
167 		Assert.assertTrue(finishOutputCalled);
168 
169 		writer.close();
170 		Assert.assertTrue(prepareSenderCalled);
171 		Assert.assertTrue(shutdownSenderCalled);
172 		Assert.assertTrue(startOutputCalled);
173 		Assert.assertTrue(finishOutputCalled);
174 
175 	}
