62  /**
63   * Writes data to <a href="http://aws.amazon.com/cloudwatch/">AWS CloudWatch</a> using the AWS Java SDK
64   *
65   * @author <a href="mailto:sascha.moellering@gmail.com">Sascha Moellering</a>
66   */
67  public class CloudWatchWriter implements OutputWriterFactory {
68  	private static final Logger log = LoggerFactory.getLogger(CloudWatchWriter.class);
69  	public static final MapEntryToDimension MAP_ENTRY_TO_DIMENSION = new MapEntryToDimension();
70  
71  	private final String namespace;
72  	private final Iterable<Map<String, Object>> dimensions;
73  
74  	private final boolean booleanAsNumber;
75  
76  	@JsonCreator
77  	public CloudWatchWriter(
78  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
79  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
80  			@JsonProperty("debug") Boolean debugEnabled,
81  			@JsonProperty("namespace") String namespace,
82  			@JsonProperty("dimensions") Collection<Map<String,Object>> dimensions,
83  			@JsonProperty("settings") Map<String, Object> settings) {
84  		this.booleanAsNumber = booleanAsNumber;
85  		this.namespace = firstNonNull(namespace, (String) settings.get("namespace"));
86  		checkArgument(!isNullOrEmpty(this.namespace), "namespace cannot be null or empty");
87  
88  		this.dimensions = firstNonNull(dimensions, (Collection<Map<String, Object>>) settings.get("dimensions"));
89  	}
90  
91  	/**
92  	 * Configuring the CloudWatch client.
93  	 *
94  	 * Credentials are loaded from the Amazon EC2 Instance Metadata Service
95  	 */
96  	private AmazonCloudWatchClient createCloudWatchClient() {
97  		AmazonCloudWatchClient cloudWatchClient = new AmazonCloudWatchClient(new InstanceProfileCredentialsProvider());
98  		cloudWatchClient.setRegion(checkNotNull(Regions.getCurrentRegion(), "Problems getting AWS metadata"));
99  		return cloudWatchClient;
100 	}
101 
102 	private ImmutableList<Dimension> createDimensions(Iterable<Map<String, Object>> dimensions) {
103 		if (dimensions == null) return ImmutableList.of();
104 
105 		return FluentIterable.from(dimensions).transform(MAP_ENTRY_TO_DIMENSION).toList();
106 	}
107 
108 	@Override
109 	public OutputWriter create() {
110 		return ResultTransformerOutputWriter.booleanToNumber(
111 				booleanAsNumber,
112 				new Writer(namespace, createCloudWatchClient(), createDimensions(dimensions))
113 		);
114 	}
115 
116 	public static class Writer extends OutputWriterAdapter {
117 
118 		@Nonnull private final String namespace;
119 		@Nonnull private final AmazonCloudWatch cloudWatchClient;
120 
121 		@Nonnull private final ObjectToDouble toDoubleConverter = new ObjectToDouble();
122 		@Nonnull private final ImmutableCollection<Dimension> dimensions;
123 
124 		public Writer(@Nonnull String namespace, @Nonnull AmazonCloudWatch cloudWatchClient, @Nonnull ImmutableCollection<Dimension> dimensions) {
125 			this.namespace = namespace;
126 			this.cloudWatchClient = cloudWatchClient;
127 			this.dimensions = dimensions;
128 		}
129 
130 		@Override
131 		public void doWrite(Server server, Query query, Iterable<Result> results) throws Exception {
132 			PutMetricDataRequest metricDataRequest = new PutMetricDataRequest();
133 			metricDataRequest.setNamespace(namespace);
134 			List<MetricDatum> metricDatumList = new ArrayList<>();
135 
136 			// Iterating through the list of query results
137 
138 			for (Result result : results) {
139 				try {
140 					metricDatumList.add(processResult(result));
141 				} catch (IllegalArgumentException iae) {
142 					log.error("Could not convert result to double", iae);
143 				}
144 			}
145 
146 			metricDataRequest.setMetricData(metricDatumList);
147 			cloudWatchClient.putMetricData(metricDataRequest);
148 		}
149 
150 		private MetricDatum processResult(Result result) {
151 			// Sometimes the attribute name and the key of the value are the same
152 			MetricDatum metricDatum = new MetricDatum();
153 			if (result.getValuePath().isEmpty()) {
154 				metricDatum.setMetricName(result.getAttributeName());
155 			} else {
156 				metricDatum.setMetricName(result.getAttributeName() + "_" + KeyUtils.getValuePathString(result));
157 			}
158 
159 			metricDatum.setDimensions(dimensions);
160 
161 			// Converts the Objects to Double-values for CloudWatch
162 			metricDatum.setValue(toDoubleConverter.apply(result.getValue()));
163 			metricDatum.setTimestamp(new Date());
164 			return metricDatum;
165 		}
166 
167 		@Override
168 		public void close() throws LifecycleException {
169 			cloudWatchClient.shutdown();
170 		}
171 	}
