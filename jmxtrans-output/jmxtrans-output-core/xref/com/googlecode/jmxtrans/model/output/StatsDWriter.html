64  /**
65   * This output writer sends data to a host/port combination in the StatsD
66   * format.
67   *
68   * @author neilh
69   */
70  @EqualsAndHashCode(exclude = {"pool", "mbean"})
71  @ToString
72  public class StatsDWriter extends BaseOutputWriter {
73  
74  	private static final Logger log = LoggerFactory.getLogger(StatsDWriter.class);
75  	public static final String ROOT_PREFIX = "rootPrefix";
76  	private static final String BUCKET_TYPE = "bucketType";
77  	private static final String STRING_VALUE_AS_KEY = "stringValuesAsKey";
78  	private static final String STRING_VALUE_DEFAULT_COUNTER = "stringValueDefaultCount";
79  	private static final Pattern STATSD_INVALID = Pattern.compile("[:|]");
80  
81  	private final ByteBuffer sendBuffer;
82  
83  	private final String bucketType;
84  	private final String rootPrefix;
85  	private final String replacementForInvalidChar;
86  	private final InetSocketAddress address;
87  	private final DatagramChannel channel;
88  	private final Boolean stringsValuesAsKey;
89  	@Nonnull private final Long stringValueDefaultCount;
90  
91  	private GenericKeyedObjectPool<SocketAddress, DatagramSocket> pool;
92  	private ManagedObject mbean;
93  
94  	@Nonnull private final ValueTransformer valueTransformer = new CPrecisionValueTransformer();

97  	/**
98  	 * Uses JmxUtils.getDefaultPoolMap()
99  	 *
100 	 * @throws IOException
101 	 */
102 	@JsonCreator
103 	public StatsDWriter(
104 			@JsonProperty("typeNames") ImmutableList<String> typeNames,
105 			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
106 			@JsonProperty("debug") Boolean debugEnabled,
107 			@JsonProperty("host") String host,
108 			@JsonProperty("port") Integer port,
109 			@JsonProperty("bucketType") String bucketType,
110 			@JsonProperty("rootPrefix") String rootPrefix,
111 			@JsonProperty(STRING_VALUE_AS_KEY) Boolean stringsValuesAsKey,
112 			@JsonProperty(STRING_VALUE_DEFAULT_COUNTER) Long stringValueDefaultCount,
113 			@JsonProperty("settings") Map<String, Object> settings,
114 			@JsonProperty("replacementForInvalidChar") String replacementForInvalidChar
115 			) throws IOException {
116 		super(typeNames, booleanAsNumber, debugEnabled, settings);
117 		log.warn("StatsDWriter is deprecated. Please use StatsDWriterFactory instead.");
118 		channel = DatagramChannel.open();
119 		sendBuffer = ByteBuffer.allocate((short) 1500);
120 
121 		// bucketType defaults to c == counter
122 		this.bucketType = firstNonNull(bucketType, (String) getSettings().get(BUCKET_TYPE), "c");
123 		this.rootPrefix = firstNonNull(rootPrefix, (String) getSettings().get(ROOT_PREFIX), "servers");
124 		// treat string attributes as key
125 		this.stringsValuesAsKey = firstNonNull(stringsValuesAsKey,
126 				(Boolean) getSettings().get(STRING_VALUE_AS_KEY), false);
127 		this.stringValueDefaultCount = firstNonNull(stringValueDefaultCount,
128 				(Long) getSettings().get(STRING_VALUE_DEFAULT_COUNTER), 1L);
129 
130 		if (host == null) {
131 			host = (String) getSettings().get(HOST);
132 		}
133 
134 		if (port == null) {
135 			port = Settings.getIntegerSetting(getSettings(), PORT, null);
136 		}
137 		this.replacementForInvalidChar = MoreObjects.firstNonNull(replacementForInvalidChar, "_");
138 
139 		checkNotNull(host, "Host cannot be null");
140 		checkNotNull(port, "Port cannot be null");
141 		this.address = new InetSocketAddress(host, port);
142 	}

145 	public void validateSetup(Server server, Query query) throws ValidationException {}

148 	@Override
149 	public void start() throws LifecycleException {
150 		try {
151 			pool = new GenericKeyedObjectPool<>(new DatagramSocketFactory());
152 			pool.setTestOnBorrow(true);
153 			pool.setMaxActive(-1);
154 			pool.setMaxIdle(-1);
155 			pool.setTimeBetweenEvictionRunsMillis(MILLISECONDS.convert(5, MINUTES));
156 			pool.setMinEvictableIdleTimeMillis(MILLISECONDS.convert(5, MINUTES));
157 
158 			this.mbean = new ManagedGenericKeyedObjectPool((GenericKeyedObjectPool) pool, "StatsdConnectionPool");
159 			ManagementFactory.getPlatformMBeanServer()
160 					.registerMBean(this.mbean, this.mbean.getObjectName());
161 		} catch (Exception e) {
162 			throw new LifecycleException(e);
163 		}
164 	}
165 
166 	@Override
167 	public void close() throws LifecycleException {
168 		try {
169 			if (this.mbean != null) {
170 				MBeanServer mbs = ManagementFactory.getPlatformMBeanServer();
171 				mbs.unregisterMBean(this.mbean.getObjectName());
172 				this.mbean = null;
173 			}
174 			if (this.pool != null) {
175 				pool.close();
176 				this.pool = null;
177 			}
178 		} catch (Exception e) {
179 			throw new LifecycleException(e);
180 		}
181 	}

183 	@Override
184 	public void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
185 
186 		List<String> typeNames = this.getTypeNames();
187 
188 		for (Result result : results) {
189 			log.debug(result.toString());
190 
191 			String key = KeyUtils.getKeyString(server, query, result, typeNames, rootPrefix);
192 			if (isNotValidValue(result.getValue())) {
193 				log.debug("Skipping message key[{}] with value: {}.", key, result.getValue());
194 				continue;
195 			}
196 
197 			// These characters can mess with formatting.
198 			String line = STATSD_INVALID.matcher(key).replaceAll(this.replacementForInvalidChar)
199 				+ computeActualValue(result.getValue()) + "|" + bucketType + "\n";
200 
201 			doSend(line.trim());
202 		}
203 	}
204 
205 	private boolean isNotValidValue(Object value){
206 		return ! (isNumeric(value) || stringsValuesAsKey);
207 	}
208 
209 	private String computeActualValue(Object value){
210 		Object transformedValue = valueTransformer.apply(value);
211 		if(isNumeric(transformedValue)){
212 			return ":" + transformedValue.toString();
213 		}
214 
215 		return "." + transformedValue.toString() + ":" + stringValueDefaultCount.toString();
216 	}
217 
218 	private synchronized boolean doSend(String stat) {
219 		try {
220 			log.debug("StatsD Message: " + stat);
221 
222 			final byte[] data = stat.getBytes("utf-8");
223 
224 			// If we're going to go past the threshold of the buffer then flush.
225 			// the +1 is for the potential '\n' in multi_metrics below
226 			if (sendBuffer.remaining() < (data.length + 1)) {
227 				flush();
228 			}
229 
230 			if (sendBuffer.position() > 0) { // multiple metrics are separated
231 				// by '\n'
232 				sendBuffer.put((byte) '\n');
233 			}
234 
235 			sendBuffer.put(data); // append the data
236 
237 			flush();
238 			return true;
239 
240 		} catch (IOException e) {
241 			log.error("Could not send metrics to Statsd", e);
242 			return false;
243 		}
244 	}
245 
246 	public synchronized boolean flush() {
247 		try {
248 			final int sizeOfBuffer = sendBuffer.position();
249 
250 			if (sizeOfBuffer <= 0) {
251 				return false;
252 			} // empty buffer
253 
254 			// send and reset the buffer
255 			sendBuffer.flip();
256 			final int nbSentBytes = channel.send(sendBuffer, this.address);
257 			sendBuffer.limit(sendBuffer.capacity());
258 			sendBuffer.rewind();
259 
260 			return sizeOfBuffer == nbSentBytes;
261 
262 		} catch (IOException e) {
263 			log.error("Could not send metrics to Statsd", e);
264 			return false;
265 		}
266 	}
