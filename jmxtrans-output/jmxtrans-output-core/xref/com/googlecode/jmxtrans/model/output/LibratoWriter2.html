23  package com.googlecode.jmxtrans.model.output;
24  
25  import com.fasterxml.jackson.core.JsonFactory;
26  import com.fasterxml.jackson.core.JsonGenerator;
27  import com.google.common.collect.ImmutableList;
28  import com.google.common.io.Closer;
29  import com.googlecode.jmxtrans.model.Query;
30  import com.googlecode.jmxtrans.model.Result;
31  import com.googlecode.jmxtrans.model.Server;
32  import com.googlecode.jmxtrans.model.naming.KeyUtils;
33  import com.googlecode.jmxtrans.model.output.support.WriterBasedOutputWriter;
34  
35  import javax.annotation.Nonnull;
36  import java.io.IOException;
37  import java.io.Writer;
38  
39  import static com.googlecode.jmxtrans.model.naming.StringUtils.cleanupStr;
40  import static com.googlecode.jmxtrans.util.NumberUtils.isNumeric;
41  import static java.util.concurrent.TimeUnit.MILLISECONDS;
42  import static java.util.concurrent.TimeUnit.SECONDS;
43  
44  public class LibratoWriter2 implements WriterBasedOutputWriter {
45  
46  	@Nonnull private final JsonFactory jsonFactory;
47  	@Nonnull private final ImmutableList<String> typeNames;
48  
49  	public LibratoWriter2(@Nonnull JsonFactory jsonFactory, @Nonnull ImmutableList<String> typeNames) {
50  		this.jsonFactory = jsonFactory;
51  		this.typeNames = typeNames;
52  	}
53  
54  	@Override
55  	public void write(@Nonnull Writer writer, @Nonnull Server server, @Nonnull Query query, @Nonnull Iterable<Result> results) throws IOException {
56  		Closer closer = Closer.create();
57  		try {
58  			JsonGenerator g = closer.register(jsonFactory.createGenerator(writer));
59  			g.writeStartObject();
60  			g.writeArrayFieldStart("counters");
61  			g.writeEndArray();
62  
63  			String source = getSource(server);
64  
65  			g.writeArrayFieldStart("gauges");
66  			for (Result result : results) {
67  				if (isNumeric(result.getValue())) {
68  					g.writeStartObject();
69  					g.writeStringField("name", KeyUtils.getKeyString(query, result, typeNames));
70  					if (source != null && !source.isEmpty()) {
71  						g.writeStringField("source", source);
72  					}
73  					g.writeNumberField("measure_time", SECONDS.convert(result.getEpoch(), MILLISECONDS));
74  					Object value = result.getValue();
75  					if (value instanceof Integer) {
76  						g.writeNumberField("value", (Integer) value);
77  					} else if (value instanceof Long) {
78  						g.writeNumberField("value", (Long) value);
79  					} else if (value instanceof Float) {
80  						g.writeNumberField("value", (Float) value);
81  					} else if (value instanceof Double) {
82  						g.writeNumberField("value", (Double) value);
83  					}
84  					g.writeEndObject();
85  				}
86  			}
87  			g.writeEndArray();
88  			g.writeEndObject();
89  			g.flush();
90  		} catch (Throwable t) {
91  			throw closer.rethrow(t);
92  		} finally {
93  			closer.close();
94  		}
95  	}
96  
97  	private String getSource(Server server) {
98  		if (server.getAlias() != null) {
99  			return server.getAlias();
100 		} else {
101 			return cleanupStr(server.getHost());
102 		}
103 	}
104 
105 }
