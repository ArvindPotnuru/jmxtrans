56  /**
57   * Feed data directly into elastic.
58   *
59   * @author Peter Paul Bakker - pp@stokpop.nl
60   */
61  
62  @NotThreadSafe
63  public class ElasticWriter extends BaseOutputWriter {
64  
65  	private static final Logger log = LoggerFactory.getLogger(ElasticWriter.class);
66  
67  	private static final String DEFAULT_ROOT_PREFIX = "jmxtrans";
68  	private static final String ELASTIC_TYPE_NAME = "jmx-entry";
69  
70  	private static final Object CREATE_MAPPING_LOCK = new Object();
71  
72  	// injected for mockito unit tests: do not make final
73  	private JestClient jestClient;
74  
75  	private final String rootPrefix;
76  	private final String connectionUrl;
77  	private final String indexName;
78  
79  	@JsonCreator
80  	public ElasticWriter(
81  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
82  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
83  			@JsonProperty("rootPrefix") String rootPrefix,
84  			@JsonProperty("debug") Boolean debugEnabled,
85  			@JsonProperty("connectionUrl") String connectionUrl,
86  			@JsonProperty("settings") Map<String, Object> settings) throws IOException {
87  
88  		super(typeNames, booleanAsNumber, debugEnabled, settings);
89  
90  		this.rootPrefix = firstNonNull(
91  						rootPrefix,
92  						(String) getSettings().get("rootPrefix"),
93  						DEFAULT_ROOT_PREFIX);
94  
95  		this.connectionUrl = connectionUrl;
96  		this.indexName = this.rootPrefix + "_jmx-entries";
97  		this.jestClient = createJestClient(connectionUrl);
98  	}
99  
100 	private JestClient createJestClient(String connectionUrl) {
101 		log.info("Create a jest elastic search client for connection url [{}]", connectionUrl);
102 		JestClientFactory factory = new JestClientFactory();
103 		factory.setHttpClientConfig(
104 				new HttpClientConfig.Builder(connectionUrl)
105 						.multiThreaded(true)
106 						.build());
107 		return factory.getObject();
108 	}
109 
110 	@Override
111 	protected void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
112 
113 		for (Result result : results) {
114 			log.debug("Query result: [{}]", result);
115 			if (isNumeric(result.getValue())) {
116 				Map<String, Object> map = new HashMap<>();
117 				map.put("serverAlias", server.getAlias());
118 				map.put("server", server.getHost());
119 				map.put("port", server.getPort());
120 				map.put("objDomain", result.getObjDomain());
121 				map.put("className", result.getClassName());
122 				map.put("typeName", result.getTypeName());
123 				map.put("attributeName", result.getAttributeName());
124 				map.put("valuePath", Joiner.on('/').join(result.getValuePath()));
125 				map.put("keyAlias", result.getKeyAlias());
126 				map.put("value", Double.parseDouble(result.getValue().toString()));
127 				map.put("timestamp", result.getEpoch());
128 
129 				log.debug("Insert into Elastic: Index: [{}] Type: [{}] Map: [{}]", indexName, ELASTIC_TYPE_NAME, map);
130 				Index index = new Index.Builder(map).index(indexName).type(ELASTIC_TYPE_NAME).build();
131 				JestResult addToIndex = jestClient.execute(index);
132 				if (!addToIndex.isSucceeded()) {
133 					throw new ElasticWriterException(String.format("Unable to write entry to elastic: %s", addToIndex.getErrorMessage()));
134 				}
135 			} else {
136 				log.warn("Unable to submit non-numeric value to Elastic: [{}] from result [{}]", result.getValue(), result);
137 			}
138 		}
139 	}
140 
141 	private static void createMappingIfNeeded(JestClient jestClient, String indexName, String typeName) throws ElasticWriterException, IOException {
142 		synchronized (CREATE_MAPPING_LOCK) {
143 			IndicesExists indicesExists = new IndicesExists.Builder(indexName).build();
144 			boolean indexExists = jestClient.execute(indicesExists).isSucceeded();
145 
146 			if (!indexExists) {
147 
148 				CreateIndex createIndex = new CreateIndex.Builder(indexName).build();
149 				jestClient.execute(createIndex);
150 
151 				URL url = ElasticWriter.class.getResource("/elastic-mapping.json");
152 				String mapping = Resources.toString(url, Charsets.UTF_8);
153 
154 				PutMapping putMapping = new PutMapping.Builder(indexName, typeName,mapping).build();
155 
156 				JestResult result = jestClient.execute(putMapping);
157 				if (!result.isSucceeded()) {
158 					throw new ElasticWriterException(String.format("Failed to create mapping: %s", result.getErrorMessage()));
159 				}
160 				else {
161 					log.info("Created mapping for index {}", indexName);
162 				}
163 			}
164 		}
165 	}
166 
167 	@Override
168 	public void start() throws LifecycleException {
169 		super.start();
170 		try {
171 			createMappingIfNeeded(jestClient, indexName, ELASTIC_TYPE_NAME);
172 		} catch (Exception e) {
173 			throw new LifecycleException("Failed to create elastic mapping.", e);
174 		}
175 	}
176 
177 	@Override
178 	public void close() throws LifecycleException {
179 		super.close();
180 		jestClient.shutdownClient();
181 	}
182 
183 	@Override
184 	public void validateSetup(Server server, Query query) throws ValidationException {
185 		// no validations
186 	}
187 
188 	@Override
189 	public String toString() {
190 		final StringBuilder sb = new StringBuilder("ElasticWriter{");
191 		sb.append("rootPrefix='").append(rootPrefix).append('\'');
192 		sb.append(", connectionUrl='").append(connectionUrl).append('\'');
193 		sb.append(", indexName='").append(indexName).append('\'');
194 		sb.append('}');
195 		return sb.toString();
196 	}
