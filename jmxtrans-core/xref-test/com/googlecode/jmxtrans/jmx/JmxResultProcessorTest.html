67  		Attribute integerAttribute = new Attribute("StartTime", 51L);
68  		ObjectInstance runtime = getRuntime();
69  		List<Result> results = new JmxResultProcessor(
70  				dummyQueryWithResultAlias(),
71  				runtime,
72  				ImmutableList.of(integerAttribute),
73  				runtime.getClassName(),
74  				TEST_DOMAIN_NAME).getResults();
75  
76  		assertThat(results).hasSize(1);
77  		Result integerResult = results.get(0);

79  		assertThat(integerResult.getAttributeName()).isEqualTo("StartTime");

81  		assertThat(integerResult.getKeyAlias()).isEqualTo("resultAlias");
82  		assertThat(integerResult.getTypeName()).isEqualTo("type=Runtime");
83  		assertThat(integerResult.getValues()).hasSize(1);
84  	}

181 	@Test(timeout = 1000)
182 	public void canReadFieldsOfTabularData() throws MalformedObjectNameException, AttributeNotFoundException, MBeanException,
183 			ReflectionException, InstanceNotFoundException {
184 		// Need to induce a GC for the attribute below to be populated
185 		Runtime.getRuntime().gc();
186 
187 		ObjectInstance runtime = null;
188 		try {
189 			runtime = getG1YoungGen();
190 		} catch (InstanceNotFoundException e) {
191 			// ignore test if G1 not enabled
192 			assumeNoException("G1 GC in Java 7/8 needs to be enabled with -XX:+UseG1GC", e);
193 		}
194 
195 		AttributeList attr;
196 		// but takes a non-deterministic amount of time for LastGcInfo to get populated
197 		while (true) { // but bounded by Test timeout
198 			attr = ManagementFactory.getPlatformMBeanServer().getAttributes(
199 					runtime.getObjectName(), new String[]{"LastGcInfo"});
200 			if (((Attribute) attr.get(0)).getValue() != null) {
201 				break;
202 			}
203 		}
204 
205 		List<Result> results = new JmxResultProcessor(
206 				dummyQueryWithResultAlias(),
207 				runtime,
208 				attr.asList(),
209 				runtime.getClassName(),
210 				TEST_DOMAIN_NAME).getResults();
211 
212 		assertThat(results.size()).isGreaterThan(2);
213 
214 		Optional<Result> result = from(results).firstMatch(new ByAttributeName("LastGcInfo"));
215 		assertThat(result.isPresent()).isTrue();
216 		// Should have primitive typed fields
217 		assertThat(result.get().getValues().size()).isGreaterThan(0);
218 		assertThat(result.get().getValues().get("duration")).isNotNull();
219 		// assert tabular fields are excluded
220 		assertThat(result.get().getValues().get("memoryUsageBeforeGc")).isNull();
221 		assertThat(result.get().getValues().get("memoryUsageAfterGc")).isNull();
222 	}

307 	private static class ByAttributeName implements Predicate<Result> {
308 		private final String attributeName;
309 
310 		public ByAttributeName(String attributeName) {
311 			this.attributeName = attributeName;
312 		}
313 
314 		@Override
315 		public boolean apply(@Nullable Result result) {
316 			if (result == null) {
317 				return false;
318 			}
319 			return attributeName.equals(result.getAttributeName());
320 		}
