23  package com.googlecode.jmxtrans.model;
24  
25  import com.fasterxml.jackson.databind.annotation.JsonSerialize;
26  import com.google.common.collect.ImmutableMap;
27  import com.googlecode.jmxtrans.model.naming.typename.TypeNameValue;
28  import lombok.EqualsAndHashCode;
29  import lombok.Getter;
30  import lombok.ToString;
31  
32  import javax.annotation.Nonnull;
33  import javax.annotation.concurrent.Immutable;
34  import javax.annotation.concurrent.ThreadSafe;
35  import java.util.Map;
36  
37  import static com.fasterxml.jackson.databind.annotation.JsonSerialize.Inclusion.NON_NULL;
38  import static com.google.common.collect.ImmutableMap.copyOf;
39  
40  /**
41   * Represents the result of a query.
42   *
43   * @author jon
44   */
45  @JsonSerialize(include = NON_NULL)
46  @ThreadSafe
47  @Immutable
48  @EqualsAndHashCode
49  @ToString
50  public class Result {
51  	@Getter private final String attributeName;
52  	@Getter private final String className;
53  	@Getter private final String objDomain;
54  	@Getter private final String typeName;
55  	@Nonnull @Getter private final ImmutableMap<String, Object> values;
56  	@Getter private final long epoch;
57  	/** Specified as part of the query. */
58  	@Getter private final String keyAlias;
59  
60  	public Result(
61  			long epoch, String attributeName, String className, String objDomain, String keyAlias, String typeName,
62  			@Nonnull Map<String, Object> values) {
63  		this.className = className;
64  		this.objDomain = objDomain;
65  		this.typeName = typeName;
66  		this.values = copyOf(values);
67  		this.epoch = epoch;
68  		this.attributeName = attributeName;
69  		this.keyAlias = keyAlias;
70  	}
71  
72  	/**
73  	 * Get typeName split into a Map
74       */
75  	public Map<String, String> getTypeNameMap() {
76  		return TypeNameValue.extractMap(this.typeName);
77  	}
78  }
