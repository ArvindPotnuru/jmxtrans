90  	/**
91  	 * The list of type names used in a JMX bean string when querying with a
92  	 * wildcard which is used to expose the actual type name value to the key
93  	 * string. e.g. for this JMX name
94  	 * <p/>
95  	 * typeName=name=PS Eden Space,type=MemoryPool
96  	 * <p/>
97  	 * If you add a typeName("name"), then it'll retrieve 'PS Eden Space' from
98  	 * the string.
99  	 * <p>
100 	 * The order of the elements of this set matches the order provided by the
101 	 * user.
102 	 */
103 	@Getter private final ImmutableSet<String> typeNames;

106 	 * The alias allows you to specify what you would like the results of the
107 	 * query to go into.

170 	private Query(
171 			String obj,
172 			List<String> keys,
173 			List<String> attr,
174 			Set<String> typeNames,
175 			String resultAlias,
176 			boolean useObjDomainAsKey,
177 			boolean allowDottedKeys,
178 			boolean useAllTypeNames,
179 			List<OutputWriterFactory> outputWriterFactories,
180 			List<OutputWriter> outputWriters
181 	) {
182 		try {
183 			this.objectName = new ObjectName(obj);
184 		} catch (MalformedObjectNameException e) {
185 			throw new IllegalArgumentException("Invalid object name: " + obj, e);
186 		}
187 		this.attr = copyOf(firstNonNull(attr, Collections.<String>emptyList()));
188 		this.resultAlias = resultAlias;
189 		this.useObjDomainAsKey = firstNonNull(useObjDomainAsKey, false);
190 		this.keys = copyOf(firstNonNull(keys, Collections.<String>emptyList()));
191 		this.allowDottedKeys = allowDottedKeys;
192 		this.useAllTypeNames = useAllTypeNames;
193 		this.outputWriters = copyOf(firstNonNull(outputWriterFactories, ImmutableList.<OutputWriterFactory>of()));
194 		// We need to preserve the order of typeNames. So note that copyOf
195 		// does not mess with the order. 
196 		this.typeNames = ImmutableSet.copyOf(firstNonNull(typeNames, Collections.<String>emptySet()));
197 
198 		this.typeNameValuesStringBuilder = makeTypeNameValuesStringBuilder();
199 
200 		this.outputWriterInstances = copyOf(firstNonNull(outputWriters, ImmutableList.<OutputWriter>of()));
201 	}

203 	public String makeTypeNameValueString(List<String> typeNames, String typeNameStr) {
204 		return this.typeNameValuesStringBuilder.build(typeNames, typeNameStr);
205 	}

211 	public Iterable<Result> fetchResults(MBeanServerConnection mbeanServer, ObjectName queryName) throws InstanceNotFoundException, IntrospectionException, ReflectionException, IOException {
212 		ObjectInstance oi = mbeanServer.getObjectInstance(queryName);
213 
214 		List<String> attributes;
215 		if (attr.isEmpty()) {
216 			attributes = new ArrayList<>();
217 			MBeanInfo info = mbeanServer.getMBeanInfo(queryName);
218 			for (MBeanAttributeInfo attrInfo : info.getAttributes()) {
219 				attributes.add(attrInfo.getName());
220 			}
221 		} else {
222 			attributes = attr;
223 		}
224 
225 		try {
226 			if (!attributes.isEmpty()) {
227 				logger.debug("Executing queryName [{}] from query [{}]", queryName.getCanonicalName(), this);
228 
229 				AttributeList al = mbeanServer.getAttributes(queryName, attributes.toArray(new String[attributes.size()]));
230 
231 				return new JmxResultProcessor(this, oi, al.asList(), oi.getClassName(), queryName.getDomain()).getResults();
232 			}
233 		} catch (UnmarshalException ue) {
234 			if ((ue.getCause() != null) && (ue.getCause() instanceof ClassNotFoundException)) {
235 				logger.debug("Bad unmarshall, continuing. This is probably ok and due to something like this: "
236 						+ "http://ehcache.org/xref/net/sf/ehcache/distribution/RMICacheManagerPeerListener.html#52", ue.getMessage());
237 			} else {
238 				throw ue;
239 			}
240 		}
241 		return ImmutableList.of();
242 	}
243 
244 	private TypeNameValuesStringBuilder makeTypeNameValuesStringBuilder() {
245 		String separator = isAllowDottedKeys() ? "." : TypeNameValuesStringBuilder.DEFAULT_SEPARATOR;
246 		Set<String> typeNames = getTypeNames();
247 		if (isUseAllTypeNames()) {
248 			return new UseAllTypeNameValuesStringBuilder(separator);
249 		} else if (typeNames != null && !typeNames.isEmpty()) {
250 			return new PrependingTypeNameValuesStringBuilder(separator, new ArrayList<>(typeNames));
251 		} else {
252 			return new TypeNameValuesStringBuilder(separator);
253 		}
254 	}

264 	public void runOutputWritersForQuery(Server server, Iterable<Result> results) throws Exception {
265 		for (OutputWriter writer : getOutputWriterInstances()) {
266 			writer.doWrite(server, this, results);
267 		}
268 		logger.debug("Finished running outputWriters for query: {}", this);
269 	}

271 	@NotThreadSafe
272 	@Accessors(chain = true)
273 	public static final class Builder {
274 		@Setter private String obj;
275 		private final List<String> attr = newArrayList();
276 		@Setter private String resultAlias;
277 		private final List<String> keys = newArrayList();
278 		@Setter private boolean useObjDomainAsKey;
279 		@Setter private boolean allowDottedKeys;
280 		@Setter private boolean useAllTypeNames;
281 		private final List<OutputWriterFactory> outputWriterFactories = newArrayList();
282 		private final List<OutputWriter> outputWriters = newArrayList();
283 		// We need to pick an order preserving Set implementation here to
284 		// avoid unpredictable ordering of typeNames.
285 		private final Set<String> typeNames = newLinkedHashSet();
286 
287 		private Builder() {}
288 
289 		/** This builder does NOT copy output writers from the given query. */
290 		private Builder(Query query) {
291 			this.obj = query.objectName.toString();
292 			this.attr.addAll(query.attr);
293 			this.resultAlias = query.resultAlias;
294 			this.keys.addAll(query.keys);
295 			this.useObjDomainAsKey = query.useObjDomainAsKey;
296 			this.allowDottedKeys = query.allowDottedKeys;
297 			this.useAllTypeNames = query.useAllTypeNames;
298 			this.typeNames.addAll(query.typeNames);
299 		}
300 
301 		public Builder addAttr(String... attr) {
302 			this.attr.addAll(asList(attr));
303 			return this;
304 		}
305 
306 		public Builder addKey(String keys) {
307 			return addKeys(keys);
308 		}
309 
310 		public Builder addKeys(String... keys) {
311 			this.keys.addAll(asList(keys));
312 			return this;
313 		}
314 
315 		public Builder addOutputWriterFactory(OutputWriterFactory outputWriterFactory) {
316 			return addOutputWriterFactories(outputWriterFactory);
317 		}
318 
319 		public Builder addOutputWriterFactories(OutputWriterFactory... outputWriterFactories) {
320 			this.outputWriterFactories.addAll(asList(outputWriterFactories));
321 			return this;
322 		}
323 
324 		public Builder addOutputWriters(Collection<OutputWriter> outputWriters) {
325 			this.outputWriters.addAll(outputWriters);
326 			return this;
327 		}
328 
329 		public Builder setTypeNames(Collection<String> typeNames) {
330 			this.typeNames.addAll(typeNames);
331 			return this;
332 		}
333 
334 		public Query build() {
335 			if (!outputWriterFactories.isEmpty()) {
336 				return new Query(
337 						this.obj,
338 						this.keys,
339 						this.attr,
340 						this.typeNames,
341 						this.resultAlias,
342 						this.useObjDomainAsKey,
343 						this.allowDottedKeys,
344 						this.useAllTypeNames,
345 						this.outputWriterFactories
346 				);
347 			}
348 			return new Query(
349 					this.obj,
350 					this.keys,
351 					this.attr,
352 					this.typeNames,
353 					this.resultAlias,
354 					this.useObjDomainAsKey,
355 					this.allowDottedKeys,
356 					this.useAllTypeNames,
357 					copyOf(this.outputWriters)
358 			);
359 		}
